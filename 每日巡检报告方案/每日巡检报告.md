```yaml
[root@AutoDeploy nginx]# cat nginxX.yml 
---
- name: 完整巡检报告 - Nginx中间件
  hosts: 0083_ngxcashier
  become: yes
  gather_facts: yes

  vars:
    nginx_bin: "/opt/nginx/sbin/nginx"
    nginx_conf: "/opt/nginx/nginx.conf"
    error_log: "/opt/nginx/log/nginx/error.log"
    access_log: "/opt/nginx/log/nginx/access.log"
    domain: "cashier.sanfu.com"
    expected_version: ["1.26.2", "1.15.2"]
    expected_ports: ["80", "443"]

    threshold:
      active_conn_warn_pct: 80
      active_conn_crit_pct: 90
      time_wait_warn: 20000
      time_wait_crit: 50000
      worker_processes: "{{ ansible_processor_count | default(1) | int }}"
      worker_connections_min: 1024
      keepalive_timeout_min: 65
      keepalive_timeout_max: 75
      sendfile: "on"
      gzip_enabled: true
      open_file_cache_enabled: true
      severe_error_keywords: ["crit", "alert", "emerg"]
      access_log_size_warn_mb: 1000
      error_4xx_warn_pct: 5
      error_5xx_warn_pct: 1
      slow_request_warn_pct: 5
      malicious_keywords: ["sql", "union", "select", "script"]
      ssl_valid_months_ok: 3
      ssl_valid_months_warn: 1

  tasks:
    # ──────────────── 1. Nginx 基础状态检查 ────────────────
    - name: 检查 Nginx Master 和 Worker 进程
      shell: "pgrep -f 'nginx: master process' && pgrep -f 'nginx: worker process' || echo 'not running'"
      register: process_status
      changed_when: false
      ignore_errors: true

    - name: 检查域名 SSL 证书剩余天数
      shell: >-
        echo | openssl s_client -connect {{ domain }}:443 -servername {{ domain }} 2>/dev/null
        | openssl x509 -noout -dates 2>/dev/null
        | grep notAfter | cut -d= -f2 | xargs -I {} date -d '{}' +%s || echo 0
      register: ssl_expiry_epoch
      changed_when: false
      ignore_errors: true

    - name: 计算 SSL 剩余天数（强制整数）
      set_fact:
        ssl_days_left: "{{ (((ssl_expiry_epoch.stdout | default('0') | int) - (ansible_date_time.epoch | default(0) | int)) / 86400) | int(0) }}"

    - name: 检查 Nginx 配置语法
      command: "{{ nginx_bin }} -t"
      register: config_syntax
      changed_when: false
      ignore_errors: true

    - name: 获取 Nginx 监听端口
      shell: "{{ nginx_bin }} -T 2>/dev/null | grep listen | grep -oP '\\d+' | sort -u | tr '\n' ',' || echo 'none'"
      register: listening_ports
      changed_when: false
      ignore_errors: true

    - name: 获取 Nginx 版本
      command: "{{ nginx_bin }} -v"
      register: nginx_version_raw
      changed_when: false
      ignore_errors: true

    - name: 解析 Nginx 版本
      set_fact:
        nginx_version: "{{ nginx_version_raw.stderr | regex_search('nginx version: nginx/([\\d.]+)', '\\1') | first | default('unknown') }}"

    # worker_processes 等配置检查（合并重复任务）
    - name: 获取 worker_processes
      shell: "{{ nginx_bin }} -T 2>/dev/null | grep worker_processes | awk '{print $2}' | tr -d ';' || echo 1"
      register: worker_proc_raw
      changed_when: false
      ignore_errors: true

    - name: 获取 worker_connections
      shell: "{{ nginx_bin }} -T 2>/dev/null | grep worker_connections | awk '{print $2}' | tr -d ';' || echo 1024"
      register: worker_conn_raw
      changed_when: false
      ignore_errors: true

    - name: 设置 worker_connections 数值
      set_fact:
        worker_connections: "{{ worker_conn_raw.stdout | trim | int(1024) }}"

    - name: 获取 keepalive_timeout
      shell: "{{ nginx_bin }} -T 2>/dev/null | grep keepalive_timeout | head -1 | awk '{print $2}' | tr -d 's;' || echo 0"
      register: keepalive_timeout_raw
      changed_when: false
      ignore_errors: true

    - name: 检查 sendfile
      shell: "{{ nginx_bin }} -T 2>/dev/null | grep -q 'sendfile on' && echo 'on' || echo 'off'"
      register: sendfile_raw
      changed_when: false
      ignore_errors: true

    - name: 检查 gzip 是否启用
      shell: "{{ nginx_bin }} -T 2>/dev/null | grep -q '^\\s*gzip on' && echo 'on' || echo 'off'"
      register: gzip_raw
      changed_when: false
      ignore_errors: true

    - name: 检查 open_file_cache 是否启用
      shell: "{{ nginx_bin }} -T 2>/dev/null | grep -q 'open_file_cache' && echo 'enabled' || echo 'disabled'"
      register: open_file_cache_raw
      changed_when: false
      ignore_errors: true

    # ──────────────── 2. 连接与请求检查 ────────────────
    - name: 统计活跃连接数（80端口示例）
      shell: "ss -ant | grep -c ':80.*ESTAB' || echo 0"
      register: active_conn
      changed_when: false
      ignore_errors: true

    - name: 统计 TIME_WAIT 连接数（80端口示例）
      shell: "ss -ant | grep -c ':80.*TIME-WAIT' || echo 0"
      register: time_wait_conn
      changed_when: false
      ignore_errors: true

    - name: 尝试获取 nginx stub_status
      uri:
        url: "http://127.0.0.1/nginx_status"
        return_content: yes
        timeout: 5
        status_code: [200, 404, 403, 500]
      register: stub_status
      ignore_errors: true
      failed_when: false

    - name: 安全解析当前活跃连接数
      set_fact:
        current_conn_raw: >-
          {% if stub_status is defined and stub_status.status == 200 and stub_status.content is defined and stub_status.content | length > 20 %}
            {% set match_result = stub_status.content | regex_findall('Active connections:\\s*(\\d+)', ignorecase=True) %}
            {{ match_result[0] | int if match_result | length > 0 else 0 }}
          {% else %}
            0
          {% endif %}

    - name: 当前连接数显示文本
      set_fact:
        current_conn_display: >-
          {% if current_conn_raw | int > 0 %}
            {{ current_conn_raw }} （stub_status 正常）
          {% else %}
            N/A （stub_status 未开启，状态码 {{ stub_status.status | default('未知') }}）
          {% endif %}

    - name: 当前连接数状态与符号
      set_fact:
        current_conn_status: "{{ '正常' if current_conn_raw | int < 20000 else '异常' }}"
        current_conn_symbol: "{{ '✅' if current_conn_raw | int < 20000 else '❌' }}"

    # ──────────────── 4. 日志监控检查 ────────────────
    - name: 检查严重错误日志
      shell: >-
        tail -n 100 {{ error_log }} 2>/dev/null
        | grep -E -i '{{ threshold.severe_error_keywords | join('|') }}'
        | wc -l || echo 0
      register: severe_errors
      changed_when: false
      ignore_errors: true

    - name: 访问日志大小 MB
      shell: "du -m {{ access_log }} 2>/dev/null | cut -f1 || echo 0"
      register: access_log_size_mb
      changed_when: false
      ignore_errors: true

    - name: 4xx 错误率（最近1000行）
      shell: >-
        tail -n 1000 {{ access_log }} 2>/dev/null
        | awk '{if ($9 ~ /^4[0-9]{2}/) count++} END {printf "%.2f", (count/NR)*100}' || echo 0.00
      register: error_4xx_pct
      changed_when: false
      ignore_errors: true

    - name: 5xx 错误率（最近1000行）
      shell: >-
        tail -n 1000 {{ access_log }} 2>/dev/null
        | awk '{if ($9 ~ /^5[0-9]{2}/) count++} END {printf "%.2f", (count/NR)*100}' || echo 0.00
      register: error_5xx_pct
      changed_when: false
      ignore_errors: true

    - name: 慢请求率 (>1s，最近1000行)
      shell: >-
        tail -n 1000 {{ access_log }} 2>/dev/null
        | awk '{if ($NF > 1) count++} END {printf "%.2f", (count/NR)*100}' || echo 0.00
      register: slow_req_pct
      changed_when: false
      ignore_errors: true

    - name: 疑似恶意请求（最近100行）
      shell: >-
        tail -n 100 {{ access_log }} 2>/dev/null
        | grep -E -i '{{ threshold.malicious_keywords | join('|') }}'
        | wc -l || echo 0
      register: malicious_req
      changed_when: false
      ignore_errors: true

    # ──────────────── 汇总结果 ────────────────
    - set_fact:
        results: []

    - name: 收集所有检查结果
      set_fact:
        results: "{{ results + [item] }}"
      loop:
        # 基础状态
        - { category: "nginx基础状态检查", name: "进程状态", status: "{{ '正常' if 'not running' not in process_status.stdout else '异常' }}", symbol: "{{ '✅' if 'not running' not in process_status.stdout else '❌' }}", value: "{{ 'Master 和 Worker 均存在' if 'not running' not in process_status.stdout else '进程缺失' }}" }
        - { category: "nginx基础状态检查", name: "SSL证书有效期", status: "{{ '正常' if (ssl_days_left | int(0)) > (threshold.ssl_valid_months_ok | int * 30) else '警告' if (ssl_days_left | int(0)) > (threshold.ssl_valid_months_warn | int * 30) else '异常' }}", symbol: "{{ '✅' if (ssl_days_left | int(0)) > (threshold.ssl_valid_months_ok | int * 30) else '⚠️' if (ssl_days_left | int(0)) > (threshold.ssl_valid_months_warn | int * 30) else '❌' }}", value: "{{ ssl_days_left | int(0) }} 天剩余" }
        - { category: "nginx基础状态检查", name: "配置文件语法", status: "{{ '正常' if config_syntax.rc == 0 else '异常' }}", symbol: "{{ '✅' if config_syntax.rc == 0 else '❌' }}", value: "{{ 'test is successful' if config_syntax.rc == 0 else config_syntax.stderr | trim | default('N/A') }}" }
        - { category: "nginx基础状态检查", name: "监听端口", status: "{{ '正常' if expected_ports | difference(listening_ports.stdout.split(',') | map('trim') | list) | length == 0 else '异常' }}", symbol: "{{ '✅' if expected_ports | difference(listening_ports.stdout.split(',') | map('trim') | list) | length == 0 else '❌' }}", value: "{{ listening_ports.stdout | default('无') }}" }
        - { category: "nginx基础状态检查", name: "Nginx版本", status: "{{ '正常' if nginx_version in expected_version else '异常' }}", symbol: "{{ '✅' if nginx_version in expected_version else '❌' }}", value: "{{ nginx_version }}" }

        # 连接与请求
        - { category: "连接与请求检查", name: "活跃连接数",
            status: "{{ '正常' if (active_conn.stdout | int(0)) < ((worker_connections | int(0)) * (threshold.active_conn_warn_pct | int) / 100) else '警告' if (active_conn.stdout | int(0)) < ((worker_connections | int(0)) * (threshold.active_conn_crit_pct | int) / 100) else '异常' }}",
            symbol: "{{ '✅' if (active_conn.stdout | int(0)) < ((worker_connections | int(0)) * (threshold.active_conn_warn_pct | int) / 100) else '⚠️' if (active_conn.stdout | int(0)) < ((worker_connections | int(0)) * (threshold.active_conn_crit_pct | int) / 100) else '❌' }}",
            value: "{{ active_conn.stdout | int(0) }} (上限约 {{ ((worker_connections | int(0)) * (threshold.active_conn_warn_pct | int) / 100) | int }})" }
        - { category: "连接与请求检查", name: "TIME_WAIT 连接数",
            status: "{{ '正常' if (time_wait_conn.stdout | int(0)) < (threshold.time_wait_warn | int) else '警告' if (time_wait_conn.stdout | int(0)) < (threshold.time_wait_crit | int) else '异常' }}",
            symbol: "{{ '✅' if (time_wait_conn.stdout | int(0)) < (threshold.time_wait_warn | int) else '⚠️' if (time_wait_conn.stdout | int(0)) < (threshold.time_wait_crit | int) else '❌' }}",
            value: "{{ time_wait_conn.stdout | int(0) }}" }
        - { category: "连接与请求检查", name: "当前连接数 (stub_status)", status: "{{ current_conn_status | default('异常') }}", symbol: "{{ current_conn_symbol | default('❌') }}", value: "{{ current_conn_display | default('N/A') }}" }

        # 配置优化（略，同原版，略微加 int 防护）
        - { category: "配置优化检查", name: "Worker进程数", status: "{{ '正常' if (worker_proc_raw.stdout | int(0)) == threshold.worker_processes else '异常' }}", symbol: "{{ '✅' if (worker_proc_raw.stdout | int(0)) == threshold.worker_processes else '❌' }}", value: "{{ worker_proc_raw.stdout }} (建议 {{ threshold.worker_processes }})" }
        - { category: "配置优化检查", name: "连接数限制", status: "{{ '正常' if (worker_conn_raw.stdout | int(0)) >= threshold.worker_connections_min else '异常' }}", symbol: "{{ '✅' if (worker_conn_raw.stdout | int(0)) >= threshold.worker_connections_min else '❌' }}", value: "{{ worker_conn_raw.stdout }}" }
        - { category: "配置优化检查", name: "keepalive_timeout", status: "{{ '正常' if (keepalive_timeout_raw.stdout | int(0)) >= threshold.keepalive_timeout_min and (keepalive_timeout_raw.stdout | int(0)) <= threshold.keepalive_timeout_max else '异常' }}", symbol: "{{ '✅' if (keepalive_timeout_raw.stdout | int(0)) >= threshold.keepalive_timeout_min and (keepalive_timeout_raw.stdout | int(0)) <= threshold.keepalive_timeout_max else '❌' }}", value: "{{ keepalive_timeout_raw.stdout }}s" }
        - { category: "配置优化检查", name: "sendfile", status: "{{ '正常' if sendfile_raw.stdout == threshold.sendfile else '异常' }}", symbol: "{{ '✅' if sendfile_raw.stdout == threshold.sendfile else '❌' }}", value: "{{ sendfile_raw.stdout }}" }
        - { category: "配置优化检查", name: "Gzip压缩", status: "{{ '正常' if gzip_raw.stdout == 'on' else '异常' }}", symbol: "{{ '✅' if gzip_raw.stdout == 'on' else '❌' }}", value: "{{ gzip_raw.stdout }}" }
        - { category: "配置优化检查", name: "文件缓存", status: "{{ '正常' if open_file_cache_raw.stdout == 'enabled' else '异常' }}", symbol: "{{ '✅' if open_file_cache_raw.stdout == 'enabled' else '❌' }}", value: "{{ open_file_cache_raw.stdout }}" }

        # 日志监控（同原版）
        - { category: "日志监控检查", name: "严重错误日志", status: "{{ '正常' if (severe_errors.stdout | int(0)) == 0 else '异常' }}", symbol: "{{ '✅' if (severe_errors.stdout | int(0)) == 0 else '❌' }}", value: "{{ severe_errors.stdout }} 条 (crit/alert/emerg)" }
        - { category: "日志监控检查", name: "访问日志大小", status: "{{ '正常' if (access_log_size_mb.stdout | int(0)) < threshold.access_log_size_warn_mb else '异常' }}", symbol: "{{ '✅' if (access_log_size_mb.stdout | int(0)) < threshold.access_log_size_warn_mb else '❌' }}", value: "{{ access_log_size_mb.stdout }} MB" }
        - { category: "日志监控检查", name: "4xx 错误率", status: "{{ '正常' if (error_4xx_pct.stdout | float(0)) < threshold.error_4xx_warn_pct else '异常' }}", symbol: "{{ '✅' if (error_4xx_pct.stdout | float(0)) < threshold.error_4xx_warn_pct else '❌' }}", value: "{{ error_4xx_pct.stdout }}%" }
        - { category: "日志监控检查", name: "5xx 错误率", status: "{{ '正常' if (error_5xx_pct.stdout | float(0)) < threshold.error_5xx_warn_pct else '异常' }}", symbol: "{{ '✅' if (error_5xx_pct.stdout | float(0)) < threshold.error_5xx_warn_pct else '❌' }}", value: "{{ error_5xx_pct.stdout }}%" }
        - { category: "日志监控检查", name: "慢请求率 (>1s)", status: "{{ '正常' if (slow_req_pct.stdout | float(0)) < threshold.slow_request_warn_pct else '异常' }}", symbol: "{{ '✅' if (slow_req_pct.stdout | float(0)) < threshold.slow_request_warn_pct else '❌' }}", value: "{{ slow_req_pct.stdout }}%" }
        - { category: "日志监控检查", name: "疑似恶意请求", status: "{{ '正常' if (malicious_req.stdout | int(0)) == 0 else '异常' }}", symbol: "{{ '✅' if (malicious_req.stdout | int(0)) == 0 else '❌' }}", value: "{{ malicious_req.stdout }} 次" }

    # ──────────────── 输出报告 ────────────────
    # ──────────────── 计算统计与报告 ────────────────
    # 先计算计数（强制转为 int）
    - set_fact:
        total_items: "{{ (results | length) | int(0) }}"
        normal_count: "{{ (results | selectattr('status', 'equalto', '正常') | list | length) | int(0) }}"
        warn_count:   "{{ (results | selectattr('status', 'equalto', '警告') | list | length) | int(0) }}"
        error_count:  "{{ (results | selectattr('status', 'equalto', '异常') | list | length) | int(0) }}"

    # 再计算健康度（全部强制 int，防止除零，加 1 兜底）
    - set_fact:
        health_score: >-
          {{ 
            (
              (normal_count | int(0) * 100.0) / 
              (total_items | int(1)) 
            ) | round(1) 
          }}

    - name: 显示 Nginx 巡检报告
      debug:
        msg: |
          ==============================================
                {{ inventory_hostname }} - Nginx 巡检报告
          ==============================================
          时间： {{ ansible_date_time.iso8601_basic_short }}

          第一部分：巡检概览
          检查类别              检查项数  正常(✅)  警告(⚠️)  异常(❌)  健康度
          nginx基础状态检查      5         {{ results | selectattr('category', 'equalto', 'nginx基础状态检查') | selectattr('status', 'equalto', '正常') | list | length }}        {{ results | selectattr('category', 'equalto', 'nginx基础状态检查') | selectattr('status', 'equalto', '警告') | list | length }}        {{ results | selectattr('category', 'equalto', 'nginx基础状态检查') | selectattr('status', 'equalto', '异常') | list | length }}        {{ (results | selectattr('category', 'equalto', 'nginx基础状态检查') | selectattr('status', 'equalto', '正常') | list | length * 100 / 5) | round(0) }}%
          连接与请求检查        3         {{ results | selectattr('category', 'equalto', '连接与请求检查') | selectattr('status', 'equalto', '正常') | list | length }}        {{ results | selectattr('category', 'equalto', '连接与请求检查') | selectattr('status', 'equalto', '警告') | list | length }}        {{ results | selectattr('category', 'equalto', '连接与请求检查') | selectattr('status', 'equalto', '异常') | list | length }}        {{ (results | selectattr('category', 'equalto', '连接与请求检查') | selectattr('status', 'equalto', '正常') | list | length * 100 / 3) | round(0) }}%
          配置优化检查          6         {{ results | selectattr('category', 'equalto', '配置优化检查') | selectattr('status', 'equalto', '正常') | list | length }}        {{ results | selectattr('category', 'equalto', '配置优化检查') | selectattr('status', 'equalto', '警告') | list | length }}        {{ results | selectattr('category', 'equalto', '配置优化检查') | selectattr('status', 'equalto', '异常') | list | length }}        {{ (results | selectattr('category', 'equalto', '配置优化检查') | selectattr('status', 'equalto', '正常') | list | length * 100 / 6) | round(0) }}%
          日志监控检查          6         {{ results | selectattr('category', 'equalto', '日志监控检查') | selectattr('status', 'equalto', '正常') | list | length }}        {{ results | selectattr('category', 'equalto', '日志监控检查') | selectattr('status', 'equalto', '警告') | list | length }}        {{ results | selectattr('category', 'equalto', '日志监控检查') | selectattr('status', 'equalto', '异常') | list | length }}        {{ (results | selectattr('category', 'equalto', '日志监控检查') | selectattr('status', 'equalto', '正常') | list | length * 100 / 6) | round(0) }}%

          本次巡检总计          {{ total_items }}         {{ normal_count }}        {{ warn_count }}        {{ error_count }}        综合健康度： {{ health_score }}%

          第二部分：详细检查结果
          {% for result in results %}
          [{{ result.symbol }}] {{ result.category }} → {{ result.name }} : {{ result.status }}  ({{ result.value }})
          {% endfor %}

          备注：如有异常项，请检查 {{ error_log }} 和 {{ access_log }}
      delegate_to: localhost
      run_once: false

```
